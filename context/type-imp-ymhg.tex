  \definefontsynonym[Iosevka-Italic]     [name:iosevkaslabmediumitalic]

  \definefontsynonym[MonoBold]       [Iosevka-Bold]       [features=default]
  \definefontsynonym[MonoBoldItalic] [Iosevka-BoldItalic] [features=default]
